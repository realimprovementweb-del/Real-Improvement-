<script>

let dayChart
let prodChart


function save(){

let today=new Date().toISOString().split("T")[0]

let data={
dayScore:parseFloat(dayScore.value),
prodScore:parseFloat(prodScore.value)
}

localStorage.setItem(today,JSON.stringify(data))

load()

}


function load(){

drawCalendar()

let dayScores=[]
let prodScores=[]

for(let i=6;i>=0;i--){

let d=new Date()
d.setDate(d.getDate()-i)

let key=d.toISOString().split("T")[0]

let stored=localStorage.getItem(key)

if(stored){

let obj=JSON.parse(stored)

dayScores.push(obj.dayScore)
prodScores.push(obj.prodScore)

}else{

dayScores.push(0)
prodScores.push(0)

}

}

updateCharts(dayScores,prodScores)

}


function drawCalendar(){

let cal=document.getElementById("calendar")

cal.innerHTML=""

for(let i=1;i<=30;i++){

let div=document.createElement("div")
div.className="day"
div.innerText=i

cal.appendChild(div)

}

let today=new Date().getDate()

let key=new Date().toISOString().split("T")[0]

let stored=localStorage.getItem(key)

if(stored){

let obj=JSON.parse(stored)

let box=cal.children[today-1]

box.classList.add("done")

box.innerHTML=today+"<br>"+obj.dayScore+" / "+obj.prodScore

}

}


function updateCharts(dayScores,prodScores){

let avg1=dayScores.reduce((a,b)=>a+b,0)/7
let avg2=prodScores.reduce((a,b)=>a+b,0)/7

document.getElementById("weekAvg").innerText="Weekly Average: "+avg1.toFixed(1)
document.getElementById("prodAvg").innerText="Weekly Average: "+avg2.toFixed(1)

if(dayChart) dayChart.destroy()
if(prodChart) prodChart.destroy()


dayChart=new Chart(document.getElementById("dayChart"),{
type:"line",
data:{
labels:["6","5","4","3","2","1","Today"],
datasets:[{
label:"Day Quality",
data:dayScores,
borderColor:"#00ff88",
backgroundColor:"#00ff88",
tension:0.35
}]
},
options:{
scales:{
y:{
min:1,
max:10,
ticks:{
color:"#000000",
font:{
size:14,
weight:"bold"
}
},
grid:{
color:"rgba(0,0,0,0.15)"
}
},
x:{
ticks:{
color:"#000000",
font:{
size:14,
weight:"bold"
}
},
grid:{
color:"rgba(0,0,0,0.15)"
}
}
}
}
})


prodChart=new Chart(document.getElementById("prodChart"),{
type:"line",
data:{
labels:["6","5","4","3","2","1","Today"],
datasets:[{
label:"Productivity",
data:prodScores,
borderColor:"#00ff88",
backgroundColor:"#00ff88",
tension:0.35
}]
},
options:{
scales:{
y:{
min:1,
max:10,
ticks:{
color:"#000000",
font:{
size:14,
weight:"bold"
}
},
grid:{
color:"rgba(0,0,0,0.15)"
}
},
x:{
ticks:{
color:"#000000",
font:{
size:14,
weight:"bold"
}
},
grid:{
color:"rgba(0,0,0,0.15)"
}
}
}
}
})

}

load()
